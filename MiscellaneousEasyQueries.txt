===================================
ITEM Branch and fine code
===================================
SELECT
ItemRecordId 

FROM 
CircItemRecords

WHERE 
AssignedBranchID = 8 AND
FineCodeID = 5
===================================
ITEM Branch, Loan Period, and Collection
===================================
SELECT
ItemRecordId 

FROM 
CircItemRecords

WHERE 
AssignedBranchID = 8 AND
LoanPeriodCodeID = 3 AND
AssignedCollectionID NOT IN ('6','8','11','15','17') 
===================================
ITEM Branch and Shelf Location
===================================
SELECT
ItemRecordId 

FROM 
CircItemRecords

WHERE 
AssignedBranchID = 8 AND
ShelfLocationID = 8

===================================
ITEM Branch and Collection
===================================
SELECT
ItemRecordId 

FROM 
CircItemRecords

WHERE 
AssignedBranchID = 19 AND
AssignedCollectionID = 14

===================================
BIB Tag
===================================
SELECT
    BibliographicRecordID

FROM
    BibliographicTags (nolock)

WHERE
    Tagnumber <> 006
===================================
ITEM Barcodes -- specific
===================================
SELECT
ItemRecordId 

FROM 
CircItemRecords

WHERE 
Barcode IN ('34083000532313','34083000405841')

===================================
Record Statuses 
--works to pull IDs for other tables too if 
--they have the name/desc just change 
--the variables
===================================
SELECT
*
FROM
    Polaris.Polaris.RecordStatuses  WITH (NOLOCK)

===================================
ITEM Barcodes *ALL*
===================================
SELECT
cir.ItemRecordID AS "Record #",
cir.Barcode AS "Barcode"

FROM 
    Polaris.Polaris.CircItemRecords cir WITH (NOLOCK)

WHERE 
AssignedBranchID = 15


===================================
ITEM Branch/Loan/Collection NOT IN
===================================
SELECT
ItemRecordId 

FROM 
CircItemRecords

WHERE 
AssignedBranchID = 8 AND
LoanPeriodCodeID = 3 AND
AssignedCollectionID NOT IN ('6','8','11','15','17')
